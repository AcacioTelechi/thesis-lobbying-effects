\begin{frame}{Objetivo e hipóteses}
  \textbf{Objetivo:} medir o efeito do lobby sobre o \textbf{comportamento parlamentar} no PE, operacionalizado como \textbf{Atividade Legislativa (AL)} = número de perguntas parlamentares 

  \vspace{0.8em}
  \textbf{Hipóteses:}
  \begin{itemize}
    \item \textbf{H1:} Maior pressão de lobby $\Rightarrow$ maior AL no domínio.
    \item \textbf{H2:} Lobby de organizações empresariais tende a aumentar mais a AL (em comparação com outros atores).
    \item \textbf{H3:} Em temas mais salientes, lobby não empresarial tende a aumentar mais a AL (em comparação com lobby empresarial).
  \end{itemize}
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{Definições}
  \textbf{Lobby:}\\[0.3em]
  \textit{O exercício de influência sobre a atividade legislativa realizado por atores interessados em um ambiente marcado por diferenças de recursos.}

  \vspace{0.8em}
  \textbf{Comportamento parlamentar / Atividade Legislativa (AL):}\\[0.3em]
  \textit{O nível de envolvimento exibido por um parlamentar em um domínio político específico.}
  \vspace{0.5em}
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{Revisão da literatura (brevíssima)}
  \begin{itemize}
    \item Literatura sobre \textbf{efeitos do lobby}: fragmentada, resultados mistos \citep{de_figueiredo_advancing_2014, baumgartner2009lobbying, mahoney_lobbying_2007}.
    \item \textbf{Desafios de identificação:} viés de seleção, variáveis omitidas, múltiplos canais, persistência temporal.
  \end{itemize}
  \vspace{0.6em}
  \textbf{Contribuição desta tese:}
  \begin{itemize}
    \item Cadeia causal \textbf{curta}: lobby $\to$ comportamento parlamentar (não resultado de política).
    \item Dados \textbf{públicos}: reuniões registradas (Registo de Transparência do PE).
    \item Modelo com \textbf{efeitos fixos de alta dimensão} para identificação causal.
  \end{itemize}
\end{frame}

\begin{frame}{Fundamentação teórica do DAG -- três dimensões do comportamento}
  \scriptsize
  \begin{tabular}{p{0.11\textwidth}p{0.45\textwidth}p{0.38\textwidth}}
    \hline
    \textbf{Dimensão} & \textbf{Fundamentação teórica} & \textbf{Referências} \\
    \hline
    \textbf{Contexto} & Ambiente político, econômico e social: crises, opinião pública, eventos midiáticos e ciclos eleitorais intensificam mobilização de interesses e alteram a agenda legislativa, gerando correlação espúria entre lobby e comportamento. & \citep{mahoney_lobbying_2007, mahoney2008brussels, baumgartner2010agendas, jones2005politics} \\
    \hline
    \textbf{Parlamento} & Regras, agenda da liderança, prazos de tramitação e estrutura das comissões afetam quem é alvo de lobby e o comportamento (ex.: relatório em tramitação atrai lobistas ao relator e aumenta sua atividade). & \citep{marshall2010lobby, kluver2015legislative, coen2019legislative} \\
    \hline
    \textbf{País} & Sistema eleitoral e eleitorado criam incentivos (voto pessoal $\to$ visibilidade; lista fechada $\to$ disciplina). Tamanho e diversidade do eleitorado moldam \textit{vote-seeking}. & \citep{mayhew2004congress, samuels1997determinantes, carey2007competing, willumsen_electoral_2019} \\
    \hline
    \textbf{Partido} & Ideologia como regras informais; disciplina via sanções e recompensas (seleção, cargos, recursos). Variáveis latentes. & \citep{hix2004electoral, helmke2012informal, carey2007competing, bressanelli2016impact} \\
    \hline
    \textbf{Aspectos individuais} & Expertise (ocupação prévia) e experiência (mandato, cargos) influenciam atuação e \textit{career-seeking}; comitês e trajetória partidária. & \citep{burden2015personal, damgaard1980dilemma, daniel2015career, chiru2020loyal} \\
    \hline
  \end{tabular}
  \vspace{0.3em}
  \scriptsize Mecanismo causal subjacente: troca \citep{de_figueiredo_advancing_2014}
\end{frame}

\begin{frame}{Metodologia -- modelo}
  \textbf{Variáveis:}
  \begin{itemize}
    \item VD: $AL_{idt}$ = contagem de perguntas (MEP $i$, domínio $d$, mês $t$).
    \item Tratamento: $L_{idt}$ = contagem de reuniões de lobby.
  \end{itemize}
  \vspace{0.5em}
  \textbf{Modelo PPML (equação principal):}
  \begin{equation*}
    AL_{idt} = \beta L_{idt} + \gamma_{ct} + \lambda_{pt} + \theta_{dt} + X'_{it}\delta + \epsilon_{idt}
  \end{equation*}
  \small
  \begin{itemize}
    \item $\gamma_{ct}$: efeito fixo país$\times$tempo \quad $\lambda_{pt}$: efeito fixo partido$\times$tempo \quad $\theta_{dt}$: efeito fixo domínio$\times$tempo
    \item $X'_{it}$: controles (ex.: cargos de liderança)
  \end{itemize}
  \normalsize
  Inferência: erros clusterizados em domínio$\times$tempo e membro.
\end{frame}

\begin{frame}{Estatísticas descritivas}
  \textbf{Painel MEP--domínio--mês} (ago/2019--out/2024; 63 meses; 9 domínios):
  \begin{itemize}
    \item 979.209 observações; 1.727 deputados únicos.
    \item Total de reuniões: 235.638 \quad Total de perguntas: 118.238.
    \item 804 deputados receberam ao menos uma reunião (46,5\%); entre tratados: média 293 reuniões/deputado (mediana 109).
  \end{itemize}
  \vspace{0.4em}
  \textbf{Lobistas (Registo de Transparência):} 4.722 organizações; 40.343 reuniões no recorte por categoria.

\begin{frame}{Modelo e justificativa -- H1}
  \textbf{Modelo:} PPML com efeitos fixos de alta dimensão.
  \begin{equation*}
    AL_{idt} = \beta_1 L_{idt} + \beta_2 L_{idt}^2 + \gamma_{ct} + \lambda_{pt} + \theta_{dt} + X'_{it}\delta + \epsilon_{idt}
  \end{equation*}
  \small Termo quadrático $L_{idt}^2$ para testar retornos marginais decrescentes. Erros clusterizados em domínio$\times$tempo.
  \vspace{0.5em}
  \textbf{Justificativa empírica:}
  \begin{itemize}
    \item PPML adequado para dados de contagem (muitos zeros); evita viés de log em MQO.
    \item Efeitos fixos país$\times$t, partido$\times$t, domínio$\times$t absorvem confundidores; identificação na variação \textit{within} MEP ao longo do tempo.
    \item $\beta_2 < 0$ significativo $\Rightarrow$ efeito marginal decrescente com o número de reuniões.
  \end{itemize}
\end{frame}

\begin{frame}{Modelo e justificativa -- H2}
  \textbf{Etapa 1 -- Persuasão (PPML):}
  \begin{equation*}
    AL_{idt} = \beta^{(k)} L_{idt}^{(k)} + \gamma_{ct} + \lambda_{pt} + \theta_{dt} + X'_{it}\delta + \epsilon_{idt}, \quad k \in \{\text{Emp., ONGs, Outros}\}
  \end{equation*}
  \textbf{Etapa 2 -- Acesso (Binomial Negativa):} $\ell$ = lobista; $B_\ell$ = orçamento; $C_\ell$ = categoria; $W_\ell$ = controles.
  \begin{equation*}
    \mathbb{E}[\text{reuniões}_\ell\,|\,Z_\ell] = \exp\big(\alpha_0 + \alpha_1 C_\ell + \alpha_2 \ln B_\ell + \alpha_3\, C_\ell\cdot\ln B_\ell + W_\ell'\eta\big)
  \end{equation*}
  \textbf{Efeito total:} $\;\text{Efeito total}_{(k,B)} \approx \underbrace{\mathbb{E}[\text{reuniões}\,|\,k,B]}_{\text{Acesso}} \times \underbrace{\beta^{(k)}}_{\text{Persuasão}}$.
  \vspace{0.3em}
  % \small Justificativa: decomposição acesso/persuasão; BN para sobredispersão; interação permite efeito do orçamento por categoria; premissa de separabilidade.
\end{frame}

\begin{frame}{Modelo e justificativa -- H3}
  \textbf{Modelo:} PPML com interações entre reuniões por categoria ($L_{idt}^{(k)}$) e saliência do tema ($\text{Sal}_{dt}$, padronizada):
  \begin{equation*}
    AL_{idt} = \sum_{k} \big[ \beta_k L_{idt}^{(k)} + \phi_k L_{idt}^{(k)} \cdot \text{Sal}_{dt} \big] + \text{EF} + X'_{it}\delta + \epsilon_{idt}
  \end{equation*}
  \small $\beta_k$: efeito marginal quando saliência = média; $\phi_k$: como o efeito varia com a saliência. Erros clusterizados em domínio$\times$tempo.
  \vspace{0.4em}
  \textbf{Justificativa empírica:}
  \begin{itemize}
    \item Saliência = volume total de reuniões no domínio-tempo (log, padronizado); proxy de atenção agregada ao tema \citep{baumgartner2010agendas}.
    \item $\phi_{\text{NGOs}}$ menos negativo que $\phi_{\text{Empresas}}$ $\Rightarrow$ ONGs mais resilientes ao aumento de saliência (vantagem relativa sob escrutínio).
    \item Mesma estrutura de EF que H1/H2; identificação na variação \textit{within} MEP.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusão}
  \textbf{Síntese dos resultados:}
  \begin{itemize}
    \item \textbf{H1:} Lobby aumenta a AL ($\approx$ 2,5\% por reunião; retornos decrescentes; inflexão $\approx$ 12 reuniões). Atores com mais recursos sustentam níveis mais altos de lobby.
    \item \textbf{H2:} Influência total = acesso $\times$ persuasão. Abaixo de $\approx$\$8,8 M: ONGs com efeito total maior; acima de $\approx$\$40 M: empresas dominam. ONGs com maior \textit{persuasão por reunião}.
    \item \textbf{H3:} Em temas mais salientes o efeito diminui para todos; ONGs mais resilientes ao escrutínio $\Rightarrow$ relativamente mais influentes sob atenção pública.
  \end{itemize}
  \vspace{0.3em}
\end{frame}

\begin{frame}{Implicações}
  \textbf{Implicações:} Influência no PE é um ecossistema (recursos, tipo de ator, saliência). Resultados desafiam visão monolítica; escrutínio público funciona como contrapeso democrático. 
  
  \vspace{0.3em}
  \textbf{Limitações:} Mede-se influência (mudança de comportamento), não poder (agir contra preferências); AL não esgota o comportamento parlamentar; dados do Registo de Transparência.

  \vspace{0.3em}

  \vspace{0.3em}
  \small \textit{Em suma:} Recursos financeiros são um fator inegável, mas não o único; credibilidade da informação e escrutínio público moldam o comportamento dos representantes e a qualidade da democracia supranacional.
\end{frame}
